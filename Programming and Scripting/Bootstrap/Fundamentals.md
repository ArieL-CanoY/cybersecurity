# div
```
class values:
	containers:
		# container
			- fixed width, can change by applying paddings and margins
		# container-fluid
			- full width, cannot change even by applying paddings and margins


	format:
		{property}{sides}-{size}
	
	properties:
		m - margin
		p - padding

	sides:
		t - top
		b - bottom
		s - start/left
		e - end/right
		x - start/left and end/right
		y - top and bottom
		blank - all sides (x and y)
		

	Examples:
	# p-(1-5)
		- padding vertically (top and bottom) and horizontally (left and right)

	# pt-(1-5)
		- add padding on top

	# pb-(1-5)
		- add padding on bottom

	# ps-(1-5)
		- add padding on left or at the start
	
	# pe-(1-5)
		- add padding on right or at the end

	# p-auto














```
