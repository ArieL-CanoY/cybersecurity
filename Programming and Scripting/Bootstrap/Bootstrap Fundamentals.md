# div
```css
class values:
	#containers:
		# container
			- fixed width, can change by applying paddings and margins
		# container-fluid
			- full width, cannot change even by applying paddings and margins


	#format:
		{property}{sides}-{size}
	
	#properties:
		m - margin
		p - padding

		sides:
		t - top
		b - bottom
		s - start/left
		e - end/right
		x - start/left and end/right
		y - top and bottom
		blank - all sides (x and y)
		

	Examples:
		# p-(1-5)
			- padding vertically (top and bottom) and horizontally (left and right)
	
		# pt-(1-5)
			- add padding on top
	
		# pb-(1-5)
			- add padding on bottom
	
		# ps-(1-5)
			- add padding on left or at the start
		
		# pe-(1-5)
			- add padding on right or at the end
	
		# p-auto
	



	#Text:
		text font size inside div:
			fs-(1-6) lower is bigger
	
			Example:
				fs-1, fs-2, fs-3
		text weight:
			tw-bold, tw-light, tw-normal(default)

	#Item/Content alignment:
		> justify-content-<options> - horizontally position
			options:
				start - default position to the left
				center - center the content/items horizontally
				end - put the content/items to the right

		> align-items-center - vertically position
			note: can affect the width and height of the content
			options:
				start - default position 
				center - center the content/items vertically
				end - put the content/items to the bottom
	#  

	
	
	










```
